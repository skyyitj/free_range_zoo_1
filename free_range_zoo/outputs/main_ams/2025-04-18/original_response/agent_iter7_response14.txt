) -> int:

    # === Adjusting Scoring Criteria ===
    max_score = -float('inf')
    best_fire = None

    dist_temperature = 0.05  # Adjust temperature coefficient to balance distance factor

    fire_level_temperature = 0.2  # Adjust temperature coefficient to balance fire intensity level factor

    for i, (fire_position, fire_level, fire_intensity, fire_weight) in enumerate(zip(fire_pos, fire_levels, fire_intensities, fire_putout_weight)):
