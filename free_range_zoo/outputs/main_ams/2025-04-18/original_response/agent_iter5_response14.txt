def single_agent_policy(

    fire_levels: List[int],

) -> int:

    # === Scoring Fire Tasks ===
    max_score = float('-inf')
    best_fire = None
    distance_temperature = 0.05  # Lower the temperature parameter to further amplify distance score differences
    fire_intensity_temperature = 0.1  # Slightly lower temperature to enhance the score difference due to fire intensities
